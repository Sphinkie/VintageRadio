# lib/keyboard_control.py
import sys
import termios
import tty
import threading
from typing import Optional, Callable

class KeyboardController:
    def __init__(self, callback: Callable[[str], None]):
        self.callback = callback
        self.running = False
        self.thread: Optional[threading.Thread] = None

    def start(self):
        """Start listening for keyboard input in a background thread."""
        self.running = True
        self.thread = threading.Thread(target=self._listen, daemon=True)
        self.thread.start()

    def stop(self):
        """Stop the listener thread."""
        self.running = False
        if self.thread:
            self.thread.join(timeout=1.0)

    def _listen(self):
        """Read single characters from stdin in raw mode."""
        fd = sys.stdin.fileno()
        old_settings = termios.tcgetattr(fd)
        
        try:
            tty.setraw(fd)  # Disable line buffering and echo
            
            while self.running:
                ch = sys.stdin.read(1)
                if ch:
                    # Map keys to actions
                    if ch.lower() == 'n':
                        self.callback('NEXT')
                    elif ch.lower() == 'a':
                        self.callback('SKIP')
                    elif ch.lower() == 'q':
                        self.callback('QUIT')
                        
        except Exception as e:
            print(f"Keyboard error: {e}", file=sys.stderr)
        finally:
            # Restore terminal settings
            termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)
